def somar(num1, num2):
"""
Esta função recebe dois números e retorna a soma deles.
"""
resultado = num1 + num2
return resultado

# --- Exemplos de Uso ---

# 1. Somando números inteiros
a = 10
b = 5
soma_inteiros = somar(a, b)
print(f"A soma de {a} e {b} é: {soma_inteiros}")
# Saída: A soma de 10 e 5 é: 15

# 2. Somando números decimais (floats)
c = 3.5
d = 2.7
soma_decimais = somar(c, d)
print(f"A soma de {c} e {d} é: {soma_decimais}")
# Saída: A soma de 3.5 e 2.7 é: 6.2
        
# 3. Usando a função diretamente com valores
print(f"A soma de 100 e 200 é: {somar(100, 200)}")
# Saída: A soma de 100 e 200 é: 300
